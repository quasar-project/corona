56 [[nodiscard]] constexpr static auto vec4_from_hex(char const* hex) -> vec4 {
57 auto constexpr cc = [](char const c) -> i32 {
58 return (c >= '0' && c <= '9') ? c - '0' : ((c >= 'a' && c <= 'f') ? c - 'a' + 10
59 : (c >= 'A' && c <= 'F') ? c - 'A' + 10
60 : throw invalid_hex_char{} // NOLINT(*-exception-baseclass)
61 );
62 };
63 auto r = (cc(hex[1]) << 4 | cc(hex[2])) & 0xFF;
64 auto g = (cc(hex[3]) << 4 | cc(hex[4])) & 0xFF;
65 auto b = (cc(hex[5]) << 4 | cc(hex[6])) & 0xFF;
66 return palette::vec4_from_rgb(r, g, b);
67 }
