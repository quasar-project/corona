73 void ThemeProvider::refresh()
74 {
75 if(not verify_path(this->folder())) {
76 llog::error("theme folder does not exist: {}", this->folder());
77 return;
78 }
79 if(this->theme_name().empty()) {
80 llog::error("theme name is empty");
81 return;
82 }
83 this->m_names = find_theme_names(m_folder)
84 .map_error([](const string& err) { llog::error("theme provider error: {}", err); })
85 .value_or(unordered_map<string, string>());
86 if(this->m_names.empty()) {
87 llog::error("no valid themes found");
88 return;
89 }
90 if(not this->m_names.contains(string(this->theme_name())))
91 {
92 llog::warn("theme '{}' not found in folder '{}'", this->theme_name(), this->folder());
93 llog::warn("themes available: {}", fmt::join(this->m_names | ranges::views::values, ", "));
94 llog::warn("defaulting to '{}'", this->m_names.begin()->first);
95 this->set_theme_name(this->m_names.begin()->first);
96 }
97 this->m_palette.clear();
98 this->m_palette = load_palette_from_file(m_folder, m_theme_name)
99 .map_error([](const string& err) { llog::error("theme provider error: {}", err); })
100 .value_or(unordered_map<PaletteType, unordered_map<string, string>>{});
101 if(this->m_palette.empty())
102 {
103 llog::error("failed to load palette");
104 return;
105 }
106 }

132 auto ThemeProvider::color(const string_view name) const -> expected<string, string>
133 {
134 if(not this->m_palette.contains(this->m_palette_type))
135 return leaf::Err(string("palette not found"));
136 if(not this->m_palette.at(this->m_palette_type).contains(string(name)))

138 return this->m_palette.at(this->m_palette_type).at(string(name));

141 auto ThemeProvider::find_theme_names(string_view folder) -> expected<unordered_map<string, string>, string>
142 {
143 llog::debug("finding theme names in ...{}", folder.substr(folder.find_last_of("\\/") + 1));
144 unordered_map<string, string> result;
145 for(const auto& file : filesystem::directory_iterator(folder))
146 {
147 if(not file.is_regular_file() or file.path().extension() != ".toml")
148 continue;
149 ifstream stream(file.path());
150 const string contents((istreambuf_iterator(stream)), istreambuf_iterator<char>());
151 toml::parse_result table;
152 try { table = toml::parse(contents); }
153 catch(const toml::parse_error& err)
154 {
155 llog::warn("failed to parse theme file {}, reason: {}", file.path().filename().string(), err.what());
156 continue;
157 }
158 const auto name = table["meta"]["name"].value<string>();
159 if(not name.has_value())
160 {
161 llog::warn("failed to get name from theme file {}", file.path().filename().string());
162 continue;
163 }
164
165 result.emplace(name.value(), file.path().filename().string());
166 }
167 if(result.empty())
168 return leaf::Err<string>("no themes found in folder");
169 for(const auto & [name, path] : result)
170 llog::debug(" {} -> {}", name, path);
171 return result;
172 }

174 auto ThemeProvider::load_palette_from_file(const string_view folder, const string_view theme_name) const -> expected<ThemeProvider::PaletteSet, string>
175 {
176 if(not this->m_names.contains(string(theme_name)))
177 return leaf::Err(string("theme not found"));
178
179 PaletteSet result;
180 const auto path = fmt::format("{}/{}", string(folder), this->m_names.at(string(theme_name)));
181 ifstream stream(path);
182 const string contents((istreambuf_iterator(stream)), istreambuf_iterator<char>());
183 toml::parse_result table;
184 try { table = toml::parse(contents); }
185 catch(const toml::parse_error& err)
186 {
187 llog::warn("failed to parse theme file {}, reason: {}", filesystem::path(path).filename().string(), err.what());
188 return leaf::Err(string("failed to parse theme file"));
189 }
190
191 for(const auto& key : ThemeTOMLColorKeys)
192 {
193 const auto dark = table["dark"][key].value<string>();
194 const auto light = table["light"][key].value<string>();
195 if(not dark.has_value() or not light.has_value())
196 {
197 llog::warn("failed to get color '{}' from theme file {}", key, filesystem::path(path).filename().string());
198 continue;
199 }
200 result[PaletteType::Dark][string(key)] = dark.value();
201 result[PaletteType::Light][string(key)] = light.value();
202 }
203
204 for(const auto& key : ThemeTOMLConfigurationKeys)
205 {
206 const auto value = table["primary"][key].value<string>();
207 if(not value.has_value())
208 {
209 llog::warn("failed to get color '{}' from theme file {}", key, filesystem::path(path).filename().string());
210 continue;
211 }
212 const auto& name = value.value();
213 for(auto& palette : result | views::values)
214 {
215 if(not palette.contains(string(name)))
216 {
217 llog::warn("failed to get primary color '{}' from theme file {}", key, filesystem::path(path).filename().string());
218 continue;
219 }
220 palette[string(key)] = palette[string(name)];
221 }
222 }
223
224 llog::debug("searhing for color keys in theme file {}: done", filesystem::path(path).filename().string());
225 return result;
226 }
