66 [[nodiscard]] constexpr auto crc16(byte const* data, usize size) -> u16 {
67 if(data == nullptr)
68 throw std::invalid_argument("crc16: data cannot be null");
69 auto c = std::numeric_limits<u16>::max();
70 for(auto i = 0_ZU; i < size; i++) {
71 c xor_eq static_cast<u16>(data[i]); // NOLINT(*-pro-bounds-pointer-arithmetic)
72 for(auto j = 0_ZU; j < 8; j++) {
73 if(c bitand 0x0001) {
74 c >>= 1;
75 c xor_eq 0xA001;
76 } else
77 c >>= 1;
78 }
79 }
80 return c;
81 }

102 if(data == nullptr)
103 throw std::invalid_argument("crc16_ccitt: data cannot be null");
104 auto c = std::numeric_limits<u16>::max();
