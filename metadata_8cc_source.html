54 auto Metadata::from_exif_file(
55 fs::path const& image_file,
56 ExifDecodeOptions const& options
57 ) noexcept(false) -> Metadata {
58 auto ifs = std::ifstream(image_file, std::ios::binary);
59 if(not ifs)
60 throw decoding_error(image_file, "failed to open file");
61 // read first 1024 bytes
62 auto exif_data = std::vector<byte>(options.max_metadata_size);
63 ifs.read(reinterpret_cast<char*>(exif_data.data()), static_cast<std::streamsize>(exif_data.size())); // NOLINT(*-pro-type-reinterpret-cast)
64 if(static_cast<usize>(ifs.gcount()) < exif_data.size())
65 throw decoding_error(image_file, fmt::format("exif data too small: {} bytes of at least {} bytes",
66 ifs.gcount(),
67 exif_data.size())
68 );
69 return from_exif_data(image_file.stem().string(), exif_data, false, options);
70 }

72 auto Metadata::from_exif_data(
73 option<string_view> filename,
74 span<byte const> exif_data,
75 bool const relative,
76 ExifDecodeOptions const& options
77 ) noexcept(false) -> Metadata {
78 llog::trace("parsing exif data for {}", filename.value_or("raw data"));
79 if(not relative)
80 exif_data = exif_data.subspan(options.header_offset);
81 auto const header = metadata::ExifHeader::from_bytes(exif_data);
82 if(header.marker != options.metadata_marker)
83 throw decoding_error(filename.value_or("raw data"), fmt::format("invalid exif marker: 0x{:04X}, expected 0x{:04X}",
84 header.marker,
85 options.metadata_marker
86 ));
87 exif_data = exif_data.subspan(sizeof(metadata::ExifHeader));
88 if(exif_data.size() < header.size)
89 throw decoding_error(filename.value_or("raw data"), fmt::format("exif data too small: {} bytes of at least {} bytes",
90 exif_data.size(),
91 header.size
92 ));
93 auto const raw = metadata::ExifMetadata::from_bytes(exif_data);
94 auto strip_nan = [](f32 const val) -> f32 {
95 if(std::isnan(val) or std::isinf(val))
96 return 0.0F;
97 return val;
98 };
99 auto self = Metadata();
100 self.image_name_ = filename.value_or("raw data");
101 self.anchor_point_ = {
102 .latitude = raw.latitude,
103 .longitude = raw.longitude,
104 .altitude = raw.altitude
105 };
106 self.resolution_ = { strip_nan(raw.dx), strip_nan(raw.dy) };
107 self.near_edge_ = strip_nan(raw.x0);
108 self.frame_offset_ = strip_nan(raw.y0);
109 self.azimuth_ = fl::math::angle<f32>::from_degrees(strip_nan(raw.azimuth));
110 self.drift_angle_ = fl::math::angle<f32>::from_degrees(strip_nan(raw.drift_angle));
111 self.size_ = { strip_nan(raw.lx), strip_nan(raw.ly) };
112 self.arc_divergence_ = fl::math::angle<f32>::from_degrees(strip_nan(raw.div));
113 self.velocity_ = strip_nan(raw.velocity / 3.6F);
114 self.frequency_interpolation_coefficient_ = strip_nan(raw.fic);
115 self.time_shift_ = strip_nan(raw.time_offset);
116 self.time_duration_ = strip_nan(raw.time_duration);
117 self.sar_mode_ = static_cast<Mode>(raw.mode);
118 self.image_type_ = static_cast<ImageType>(raw.image_type);
119 auto const actual_crc = raw.checksum();
120 if(actual_crc != raw.crc)
121 llog::warn("exif data checksum mismatch: expected 0x{:04X}, actual 0x{:04X}", actual_crc, raw.crc);
122 self.crc_valid_ = raw.crc == actual_crc;
123 llog::debug("successfully parsed exif data from {}", self.name());
124 return self;
125 }
